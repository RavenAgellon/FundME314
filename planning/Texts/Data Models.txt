UserAccount {
	id, email, first_name, last_name, password_hash
	role_id --> links to UserRole
	status: "active" | "suspended"
	date_created
}

UserRole {
	id, role_name
	status: "active" | "suspended"

FRA (Fundraiser Activity) {
	id, title, description, goal_amount
	status: "active" | "completed" | "suspended"
	category_id --> links to Category
	created_by --> links to User (must be fundraiser)
	view_count, save_count
	created_at, updated_at
}

FRACategory{
	id, name, description
}

Favourite{
	id
	donee_id --> links to User (must be a donee)
	fra_id --> links to FRA
	saved_at
}

Report {
	id, type: "daily" | "weekly" | "mothly"
	generated_by --> links to User (platform mgmt.)
	data_json
	generated_at
}
